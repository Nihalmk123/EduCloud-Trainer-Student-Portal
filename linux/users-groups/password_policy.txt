# ========================================================
# EduCloud Password Policy (Project Requirements)
# ========================================================

## POLICY REQUIREMENTS (from project PDF)
✅ Password expiry: 30 days
✅ Minimum length: 10 characters  
✅ Complexity: 1 uppercase, 1 number, 1 special character

## IMPLEMENTATION STEPS

### 1. Edit /etc/login.defs (system-wide defaults)

PASS_MAX_DAYS 30 # Password expiry
PASS_MIN_DAYS 0 # No minimum days between changes
PASS_MIN_LEN 10 # Minimum password length
PASS_WARN_AGE 7 # Warn 7 days before expiry


### 2. Install password quality checker
sudo yum install libpwquality -y


### 3. Configure password complexity (/etc/security/pwquality.conf)
sudo nano /etc/security/pwquality.conf

minlen = 10
dcredit = -1 # At least 1 digit
ucredit = -1 # At least 1 uppercase
lcredit = -1 # At least 1 lowercase
ocredit = -1 # At least 1 special character


### 4. Apply policy to existing users
sudo chage -M 30 trainer1 trainer2 student1 student2 student3 support1
sudo chage -W 7 trainer1 trainer2 student1 student2 student3 support1


### 5. Test password policy
sudo passwd student1


## EXPECTED OUTPUT (chage -l)
Last password change : Feb 09, 2026
Password expires : Mar 11, 2026
Password inactive : never
Account expires : never
Minimum number of days between password : 0
Maximum number of days between password : 30
Number of days of warning before : 7


## VERIFICATION COMMANDS
sudo chage -l trainer1 student1 support1

getent group trainers
getent group students
getent group support

## SAMPLE PASSWORDS (for testing only)
trainer1:  `EduCloudT1!2026`
student1:  `EduCloudS1#2026`
